on:
  workflow_dispatch:

          - preview

    branches:
      - main
      - "release/**"

jobs:
  build:
    name: Build Android APK
    runs-on: ubuntu-latest
    if: ${{ github.event_name == 'push' || !inputs.ota_only }}
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: 20

      - name: Setup pnpm
        uses: pnpm/action-setup@v4
        with:
          version: 10.26.1

      - name: Setup EAS
        uses: expo/expo-github-action@v8
        with:
          eas-version: latest
          token: ${{ secrets.EXPO_TOKEN }}

      - name: Install dependencies
        run: pnpm install --no-frozen-lockfile

      - name: Push secrets to EAS environment
        working-directory: artifacts/mobile
        env:
          EXPO_TOKEN: ${{ secrets.EXPO_TOKEN }}
          GOOGLE_WEB_CLIENT_ID: ${{ secrets.GOOGLE_WEB_CLIENT_ID }}
          MAPBOX_PUBLIC_TOKEN: ${{ secrets.MAPBOX_PUBLIC_TOKEN }}
        run: |
          push_env() {
            local name="$1"
            local value="$2"
            if [ -z "$value" ]; then
              echo "Skipping $name (empty)"
              return
            fi
            echo "$value" | eas env:create --name "$name" \
              --type plaintext --scope project --non-interactive 2>/dev/null \
            || echo "$value" | eas env:update --name "$name" \
              --scope project --non-interactive 2>/dev/null \
            || echo "Warning: could not set $name in EAS"
          }
          push_env GOOGLE_WEB_CLIENT_ID "$GOOGLE_WEB_CLIENT_ID"
          push_env MAPBOX_PUBLIC_TOKEN "$MAPBOX_PUBLIC_TOKEN"

      - name: Build Android
        working-directory: artifacts/mobile
        env:
          MAPBOX_DOWNLOAD_TOKEN: ${{ secrets.MAPBOX_DOWNLOAD_TOKEN }}
        run: eas build --platform android --profile ${{ inputs.profile || 'preview' }} --non-interactive

// Firestore security rules for the Community Notes feature.
//
// This project's live rules are NOT stored in this repo — they must be
// deployed manually via the Firebase Console (Build > Firestore Database >
// Rules) or the Firebase CLI. This file is just the source of truth for the
// `community_notes` subcollection; merge this `match` block into your
// existing `rules_version = '2'` / `service cloud.firestore` document,
// alongside whatever rules already exist for `trails`, `photos`, `events`, etc.
//
// Why these rules:
// - Create: any signed-in user can post a note, but only as themselves
//   (createdBy must equal their own uid) with the required fields.
// - Update: only allows appending your OWN uid to `confirmedBy` (the
//   upvote/"still accurate" action) — no other field may change on update.
// - Delete: the author can always delete their own note. Anyone (not just
//   the author) may delete a note once it is more than 48h old, so the
//   client-side expiry cleanup in map.tsx keeps working after these rules
//   are deployed. (Alternative: skip this and set up a Firestore TTL policy
//   on `createdAt` instead, then remove the client-side delete entirely.)

match /trails/{trailId}/community_notes/{noteId} {
  allow read: if request.auth != null;

  allow create: if request.auth != null
    && request.resource.data.createdBy == request.auth.uid
    && request.resource.data.keys().hasAll(['type', 'message', 'lat', 'lng', 'createdBy', 'createdByName', 'createdAt', 'createdAtFallback', 'confirmedBy'])
    && request.resource.data.confirmedBy == [];

  allow update: if request.auth != null
    && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['confirmedBy'])
    && request.resource.data.confirmedBy == resource.data.confirmedBy.concat([request.auth.uid]);

  allow delete: if request.auth != null
    && (resource.data.createdBy == request.auth.uid
        || request.time > resource.data.createdAt + duration.value(48, 'h'));
}

// TerraPulse — Firestore Security Rules
//
// This is the SOURCE OF TRUTH for this project's Firestore rules. It is
// NOT deployed automatically — Replit cannot publish Firestore rules on
// your behalf. To apply it:
//   1. Firebase Console → Build → Firestore Database → Rules tab
//   2. Select all existing text and replace it with the contents below
//   3. Click "Publish"
// Rules take effect within ~1 minute. No app rebuild or OTA update needed.
//
// This replaces the default "test mode" rule (open read/write to everyone,
// expiring 2026-07-20) with rules scoped to how the TerraPulse app actually
// reads and writes data, covering every collection currently in use:
// trails (+ photos/events/community_notes subcollections), users (+
// gallery/rides subcollections), and live_streams.
//
// Known limitation (not fixed here, flagging for awareness): profile
// privacy (`isPublic`) is enforced client-side only — any signed-in user
// can technically read another user's doc/gallery directly via the SDK,
// the app just chooses not to display it when isPublic is false. Locking
// that down further would need a design decision (e.g. splitting private
// fields into a subcollection) — out of scope for this pass.

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // ── users/{uid} ──────────────────────────────────────────────────────
    // Profile doc: displayName, photoURL, email, isPublic, vehicleSpecs,
    // achievements, achievementDates, completedTrails. Read broadly because
    // leaderboard/community screens list all users; write restricted to
    // the owner.
    match /users/{uid} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == uid;
      allow update: if request.auth != null && request.auth.uid == uid;

      // users/{uid}/gallery/{itemId} — photo/video gallery items
      match /gallery/{itemId} {
        allow read: if request.auth != null;
        allow create, update: if request.auth != null
          && request.auth.uid == uid
          && request.resource.data.keys().hasAll(['url', 'type', 'uploadedAt']);
      }

      // users/{uid}/rides/{rideId} — private ride history, owner-only.
      // update/delete allow the rider to rename/split/delete a ride after recording.
      match /rides/{rideId} {
        allow read: if request.auth != null && request.auth.uid == uid;
        allow create: if request.auth != null && request.auth.uid == uid;
        allow update, delete: if request.auth != null && request.auth.uid == uid;

        // users/{uid}/rides/{rideId}/track/data — raw GPS breadcrumb array,
        // stored separately from the ride doc so ride-history list queries
        // don't have to download the full track on every snapshot.
        match /track/{trackDocId} {
          allow read: if request.auth != null && request.auth.uid == uid;
          allow create, delete: if request.auth != null && request.auth.uid == uid;
        }
      }

      // users/{uid}/crew/{memberId} — mutual crew list (My Garage / My Crew).
      // Owner can read their full crew list. Any authenticated user can write a
      // crew entry when creating the mutual link after a friend-request accept.
      match /crew/{memberId} {
        allow read: if request.auth != null && request.auth.uid == uid;
        allow create, update: if request.auth != null;
        allow delete: if request.auth != null && request.auth.uid == uid;
      }

      // users/{uid}/notifications/{notifId} — in-app notification center.
      // Owner reads their own notifications; any authenticated user can create
      // (friend-request sender or crew-acceptor writes into the recipient's inbox).
      // Owner can update (mark read / update status).
      match /notifications/{notifId} {
        allow read: if request.auth != null && request.auth.uid == uid;
        allow create: if request.auth != null;
        allow update: if request.auth != null && request.auth.uid == uid;
      }

      // users/{uid}/vehicles/{vehicleId} — My Garage fleet (My Rides).
      // Strictly private — only the owner reads or writes.
      match /vehicles/{vehicleId} {
        allow read, write: if request.auth != null && request.auth.uid == uid;
      }
    }

    // ── friendRequests/{requestId} ───────────────────────────────────────
    // A pending request created by the sender. The recipient accepts or
    // declines via the Notifications tab. Both parties can read their own
    // requests (query by fromUid or toUid). Only the sender creates; only
    // the recipient updates (to accepted/rejected); neither may delete.
    match /friendRequests/{requestId} {
      allow read: if request.auth != null
        && (resource.data.fromUid == request.auth.uid
            || resource.data.toUid == request.auth.uid);
      allow create: if request.auth != null
        && request.resource.data.fromUid == request.auth.uid
        && request.resource.data.keys().hasAll(['fromUid', 'toUid', 'status', 'createdAt']);
      allow update: if request.auth != null
        && resource.data.toUid == request.auth.uid
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['status'])
        && request.resource.data.status in ['accepted', 'rejected'];
    }

    // ── trails/{trailId} ─────────────────────────────────────────────────
    // Built-in trails are hardcoded client-side and have no Firestore doc;
    // this only governs user-submitted trails and the subcollections that
    // hang off any trail id (built-in or user-submitted).
    match /trails/{trailId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null
        && request.resource.data.submittedBy == request.auth.uid
        && request.resource.data.isUserSubmitted == true;

      // trails/{trailId}/photos/{photoId}
      match /photos/{photoId} {
        allow read: if request.auth != null;
        allow create: if request.auth != null
          && request.resource.data.uploadedBy == request.auth.uid
          && request.resource.data.keys().hasAll(['url', 'uploadedBy', 'createdAt']);
      }

      // trails/{trailId}/events/{eventId}
      match /events/{eventId} {
        allow read: if request.auth != null;
        allow create: if request.auth != null
          && request.resource.data.createdBy == request.auth.uid
          && request.resource.data.keys().hasAll(['title', 'eventType', 'dateStr', 'time', 'visibility', 'createdBy', 'createdByName', 'attendees', 'description', 'createdAt']);
        // RSVP join/leave: only the `attendees` array may change, and only
        // by exactly one entry at a time (join = +1, leave = -1). This does
        // not verify WHICH entry changed (Firestore rules can't easily
        // inspect array-of-map diffs) — a known, low-severity gap.
        allow update: if request.auth != null
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['attendees'])
          && (
               request.resource.data.attendees.size() == resource.data.attendees.size() + 1
               || request.resource.data.attendees.size() == resource.data.attendees.size() - 1
             );
      }

      // trails/{trailId}/contributors/{userId}
      // Anyone who helped map or verify this trail can add/remove their own doc.
      match /contributors/{userId} {
        allow read: if request.auth != null;
        allow create: if request.auth != null
          && request.auth.uid == userId
          && request.resource.data.keys().hasAll(['uid', 'displayName', 'addedAt'])
          && request.resource.data.uid == request.auth.uid;
        allow delete: if request.auth != null && request.auth.uid == userId;
      }

      // trails/{trailId}/community_notes/{noteId}
      match /community_notes/{noteId} {
        allow read: if request.auth != null;

        allow create: if request.auth != null
          && request.resource.data.createdBy == request.auth.uid
          && request.resource.data.keys().hasAll(['type', 'message', 'lat', 'lng', 'createdBy', 'createdByName', 'createdAt', 'createdAtFallback', 'confirmedBy'])
          && request.resource.data.confirmedBy == [];

        allow update: if request.auth != null
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['confirmedBy'])
          && request.resource.data.confirmedBy == resource.data.confirmedBy.concat([request.auth.uid]);

        allow delete: if request.auth != null
          && (resource.data.createdBy == request.auth.uid
              || request.time > resource.data.createdAt + duration.value(48, 'h'));
      }
    }

    // ── sos_beacons/{uid} ────────────────────────────────────────────────
    // One document per rider, keyed by their Firebase UID. Active beacons
    // (active==true) are broadcast live to every authenticated user so they
    // can appear as red pulsing markers on the map.
    // Writes are restricted to the document owner (uid == their own UID).
    // Deactivating is a subset update (active:false) — also owner-only.
    //
    // NESTING NOTE: this block must stay inside match /databases/{database}/documents.
    // Placing it outside (after the closing brace) causes silent permission denial.
    match /sos_beacons/{uid} {
      allow read: if request.auth != null;
      allow create, update: if request.auth != null && request.auth.uid == uid;
      allow delete: if request.auth != null && request.auth.uid == uid;

      // ── Chat with the SOS rider ──────────────────────────────────────
      // Any signed-in rider can read the thread and post messages (authored
      // as themselves). The beacon owner can delete any message (cleanup on
      // deactivate); authors can delete their own.
      match /messages/{messageId} {
        allow read: if request.auth != null;
        allow create: if request.auth != null
                      && request.resource.data.uid == request.auth.uid
                      && request.resource.data.text is string
                      && request.resource.data.text.size() <= 500;
        allow delete: if request.auth != null
                      && (request.auth.uid == uid
                          || request.auth.uid == resource.data.uid);
      }

      // ── "Rescue is on the way" responders ────────────────────────────
      // One doc per responder, keyed by the responder's own UID. Riders can
      // only add/remove themselves; the beacon owner can delete any (cleanup
      // on deactivate).
      match /responders/{responderUid} {
        allow read: if request.auth != null;
        allow create, update: if request.auth != null
                              && request.auth.uid == responderUid;
        allow delete: if request.auth != null
                      && (request.auth.uid == uid
                          || request.auth.uid == responderUid);
      }
    }

    // ── live_streams/{uid} ───────────────────────────────────────────────
    // One doc per streamer, keyed by their own uid.
    match /live_streams/{uid} {
      allow read: if request.auth != null;
      allow create, update, delete: if request.auth != null && request.auth.uid == uid;
    }

    // ── group_rides/{rideId} ─────────────────────────────────────────────
    // Any signed-in user may read rides (to check for an active ride on a
    // trail). Only the creator can mark a ride inactive (end it). Creating
    // a new ride is open to any signed-in user.
    match /group_rides/{rideId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null;
      allow update: if request.auth != null && request.auth.uid == resource.data.createdBy;
      allow delete: if false;

      // members/{uid} — each rider writes/deletes only their own member doc.
      // All members can read the full member list for the ride.
      match /members/{uid} {
        allow read: if request.auth != null;
        allow create, update: if request.auth != null && request.auth.uid == uid;
        allow delete: if request.auth != null && request.auth.uid == uid;
      }

      // messages/{messageId} — any ride member can post; all members can read.
      // Deletion not permitted (immutable chat log for the ride duration).
      match /messages/{messageId} {
        allow read: if request.auth != null;
        allow create: if request.auth != null
          && request.resource.data.uid == request.auth.uid
          && request.resource.data.text is string
          && request.resource.data.text.size() > 0
          && request.resource.data.text.size() <= 500;
        allow update, delete: if false;
      }
    }

  } // ← closes match /databases/{database}/documents — sos_beacons must be ABOVE this line
}
